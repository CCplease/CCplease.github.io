思考与坑点

这题应该是所有甲级测试里，关于队列模拟比较难的了，这个难不是说跟树啊图啊那种难，是比较烦，细节太多了。这题不推荐大家把所有人员放在一个固定的容器里，最好是分两个，一个普通，一个vip(可爱开始偷懒就存在一个容器里，哭了，写不出来),这样比较容易。

那么接下来我说几个需要注意的点：

  • 这题虽然是队列但是其实并不用queue，直接用成员函数erase()处理一个删一个就行
  • 题目说了每个人玩的时间不能超过2hours，所以你在输入的时候就可以处理一下

  • 题目说，如果

    AC代码

    #include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<queue>
#include<algorithm>
#include<vector>
#include<set>
#include<string>
#pragma warning(disable:4996)
using namespace std;

constexpr auto starttime = 8 * 3600;
constexpr auto  endtime = 21 * 3600;


typedef struct node
{
	int start_time,  play_time;
	bool flag;
}Node;

typedef struct table
{
	int start_time = starttime, num = 0;
	bool vip = 0;
}table;
vector<table> tab;

bool cmp(Node a, Node b)
{
	return a.start_time < b.start_time;
}

void chuli_vip(int j, int time, vector<Node> &tmp)
{
	if (!tmp.empty())
	{
		auto it = tmp.begin();
		if (it->start_time <= time && it->flag == true)			//注意小于也行
		{
			tab[j].start_time = time + it->play_time;
			tab[j].num++;
			printf("%02d:%02d:%02d", it->start_time / 3600, it->start_time / 60 % 60, it->start_time % 60);
			printf(" %02d:%02d:%02d", time / 3600, time / 60 % 60, time % 60);
			printf(" %d\n", (time - it->start_time + 30) / 60);

			tmp.erase(it);
		}
	}
}

void chuli(int j, int time, vector<Node> &tmp)
{
	if (!tmp.empty())
	{
		auto it = tmp.begin();
		if (it->start_time <= time)			//注意小于也行
		{
			tab[j].start_time = time + it->play_time;
			tab[j].num++;
			printf("%02d:%02d:%02d", it->start_time / 3600, it->start_time / 60 % 60, it->start_time % 60);
			printf(" %02d:%02d:%02d", time / 3600, time / 60 % 60, time % 60);
			printf(" %d\n", (time - it->start_time + 30) / 60);

			tmp.erase(it);

		}
	}
}

int main() {
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif

	int n;
	cin >> n;

	vector<Node> v_o;
	vector<Node> v_vip;

	int cnt_o = 0, cnt_vip = 0;
	for (int i = 0; i < n; i++)
	{
		Node tmp;
		int h, m, s, play;
		int flag = 0;
		scanf("%d:%d:%d %d %d", &h, &m, &s, &play, &flag);

		play = play > 120 ? 120 * 60 : play * 60;

		tmp.start_time = h * 3600 + m * 60 + s;
		tmp.play_time = play;
		tmp.flag = flag;
		if (flag == 1)
		{
			v_vip.push_back(tmp);
			cnt_vip++;
		}
		else
		{
			v_o.push_back(tmp);
			cnt_o++;
		}
	}

	sort(v_o.begin(), v_o.end(), cmp);
	sort(v_vip.begin(), v_vip.end(), cmp);

	int k, m;
	cin >> k >> m;

	//这里tab从1开始，所以，v[0]我们是不用的
	tab.resize(k + 1);
	for (int i = 1; i <= m; i++)
	{
		int temp;
		cin >> temp;
		tab[temp].vip = 1;
	}

	//到这里输入处理完毕，大家可以调试然后看下容器里面的内容，方便理解
	//接下来开始处理，优先处理vip桌子

	for (int i = starttime; i < endtime; i++)
	{
		for (int j = 1; j <= k; j++)
		{
			if (tab[j].start_time <= i && tab[j].vip == 1)
			{
				//说明找到了空的vip桌子，优先处理vip
				chuli_vip(j, i, v_vip);

				if (v_vip.empty())	break;
			}
		}

		//说明vip桌子不空，那么也就是说vip和普通人待遇一样
		for (int j = 1; j <= k; j++)
		{
			if (tab[j].start_time <= i)
			{
				if (v_o.size() != 0 && v_vip.size() != 0)
				{
					//说明都不为空，那就两个容器交叉比较
					//注意这个地方没有等于，题目说了
					auto it = v_o.begin(), itv = v_vip.begin();

					if (it->start_time < itv->start_time)
						chuli(j, i, v_o);
					else if (it->start_time > itv->start_time)
						chuli_vip(j, i, v_vip);
				}
				else if (v_o.size())
					chuli(j, i, v_o);
				else if (v_vip.size())
					chuli_vip(j, i, v_vip);
				else
					break;
			}
		}
		if (v_o.size() == 0 && v_vip.size() == 0)	break;
	}

	for (int i = 1; i < tab.size(); i++)
	{
		if (i == 1)
			cout << tab[i].num;
		else
			cout << " " << tab[i].num;
	}

	return 0;
}
