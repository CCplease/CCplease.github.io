题目

样例输入

Input_1

27 2

Output_1

Yes
1 1 0 1 1

Input_2

121 5

Output_2

No
4 4 1

思路和坑点

这题本身是水题没什么问题，问题在于测试点很坑，很多人一看到题目直接to_string和reverse组合输出很容易，然后就掉坑了，这题无论进制转换还是倒置数字都要手写，用循环才可以

当然啦其他题目，to_string和reverse还是可以用的

错误AC代码

按以下写，你会发现测试点2，4不过，为啥呢，给你一组数字，体验一下

689212004 256

#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<algorithm> 
#pragma warning(disable:4996)
using namespace std;

string intToA(int n, int radix)    //n是待转数字，radix是指定的进制
{
	string ans = "";
	do {
		int t = n % radix;
		if (t >= 0 && t <= 9)	ans += t + '0';
		else ans += t - 10 + 'a';
		n /= radix;
	} while (n != 0);	//使用do{}while（）以防止输入为0的情况
	reverse(ans.begin(), ans.end());
	return ans;
}

int main() {
#ifdef ONLINE_JUDGE    
#else    
	freopen("1.txt", "r", stdin);
#endif

	int a, b;

	cin >> a >> b;

	string str = intToA(a, b);

	string temp = str;

	reverse(str.begin(), str.end());

	if (temp == str)
		cout << "Yes" << endl;
	else
		cout << "No" << endl;

	for (int i = 0; i < temp.size(); i++)
	{
		if (i == 0)	cout << temp[i];
		else
			cout<<" "<<temp[i];
	}
	return 0;
}

这个错误在哪里呢？在to_string这个函数把int转为string的时候，所有我们就用int数组来输出就行，还有reverse函数也不可以用，用了你会发现错误，所以你只能用循环倒置整个数

正确AC代码

#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<map>
#include<algorithm> 
#include<vector>
#include<queue>
#include<string> 
#pragma warning(disable:4996)
using namespace std;

vector<int> convert(int n, int radix)
{
	vector<int> ans;
	do 
	{
		ans.push_back(n % radix);
		n /= radix;

	} while (n != 0);
	return ans;
}

int main() {
#ifdef ONLINE_JUDGE    
#else    
	freopen("1.txt", "r", stdin);
#endif

	int a, b;
	bool flag=true;

	cin >> a >> b;

	vector<int> ans = convert(a, b);


	for (int i = 0; i < ans.size() / 2; i++)
		if (ans[i] != ans[ans.size() - 1 - i])
			flag = false;

	if (flag)
		cout << "Yes" << endl;
	else
		cout << "No" << endl;

	for (int i = ans.size()-1;i>=0;i--)
	{
		if (i == ans.size()-1)	cout << ans[i];
		else
			cout<<" "<<ans[i];
	}
	return 0;
}
