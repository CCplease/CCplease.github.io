思路和坑点

这题总体来说没什么坑点，题目看起来很长，耐心读完你就发现思路清晰，我们来分析一下。

  • 首先，每天不同时段的收费情况要用一个int arr[25]数组保存，为啥是25呢，最后一个存储一天总共的费用。
  • 其次，人的名字+通话时段+状态，很明显结构体跑不掉
  • 因为通话记录是散的，需要排序，而且会有脱机记录匹配不成功的记录存在，所以我们想到要把每个人分类，那么就离不开map，主键是人名，键值就可以是这个人的通话情况，从on-line到off-line算一组，所以map<string,vector<结构体>>就出来了。

  • 最后就是费用计算，我们以00：00：00为基准，所有的费用都是跟他比较，然后相减。这也就是数组开辟25的作用

    #include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<map>
#include<sstream>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

typedef struct node
{
	string str;
	int month, day, hour, minute, sum_time,status;
}NODE;

bool cmp(NODE a, NODE b)
{
	return a.str != b.str ? a.str < b.str : a.sum_time < b.sum_time;
}

double bill(node call, int *rate) 
{
//比如说一个人off-line的时间是01：07：00，那么就是1天*一天总费用*60
	double total = rate[call.hour] * call.minute + rate[24] * 60 * call.day;

//然后再算一天理的费用，也就是00:00-07：00的费用
	for (int i = 0; i < call.hour; i++)
		total += rate[i] * 60;
	return total / 100.0;
}

int main()
{
#ifdef ONLINE_JUDGE                
#else    
	freopen("1.txt", "r", stdin);
#endif 
	
	int price[25] = { 0 }, n;

	for (int i = 0; i < 24; i++)
	{
		cin >> price[i];

		//这个地方，要注意，是指一天的不同时间段费用的总和
		price[24] += price[i];
	}
		
	cin >> n;
	
	vector<NODE> v(n);
	for (int i = 0; i < n; i++)
	{
		string temp;

		cin >> v[i].str;
		scanf("%d:%d:%d:%d", &v[i].month, &v[i].day, &v[i].hour, &v[i].minute);

		cin >> temp;
		temp == "on-line" ? v[i].status = 1 : v[i].status = 0;

		v[i].sum_time = v[i].day * 24 * 60 + v[i].hour * 60 + v[i].minute;
	}

	//到这里我们已经存储完毕，接下来我们用map分下类
	sort(v.begin(), v.end(), cmp);

	map<string, vector<NODE>> mmap;
	for (int i = 1; i < n; i++)
	{
		if (v[i].str == v[i - 1].str&&v[i].status == 0 && v[i - 1].status == 1)
		{
			mmap[v[i - 1].str].push_back(v[i - 1]);
			mmap[v[i].str].push_back(v[i]);
		}
	}

	for (auto it : mmap)
	{
		cout << it.first;
		printf(" %02d\n", it.second[0].month);

		double temp=0.0;
		for (size_t i = 1; i < it.second.size(); i+=2)
		{
			double total_price = bill(it.second[i], price) - bill(it.second[i - 1], price);

			printf("%02d:%02d:%02d %02d:%02d:%02d %d $%.2lf\n", it.second[i - 1].day, it.second[i - 1].hour, it.second[i - 1].minute, it.second[i].day, it.second[i].hour, it.second[i].minute,it.second[i].sum_time-it.second[i-1].sum_time,total_price);

			temp += total_price;
		}
		printf("Total amount: $%.2lf\n", temp);
	}

	system("pause");
	return 0;
}
