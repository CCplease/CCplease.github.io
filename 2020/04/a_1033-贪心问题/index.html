思考与坑点

本题照搬算法笔记

首先，我先在当前加油站考虑一个问题，看看后面的站有没有能到并且当前加油站便宜的，如果有我们就到那个站去，注意，只是选择了目标加油站，不代表你选择了，就要去那个站加油，你到了那个站，还要看看下面有没有能到且更便宜的，同时还要看当前油箱有没有油

AC代码

#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
#include<map>
#include<unordered_map>
#include<queue>
#include<set>
#include<stack>
#pragma warning (disable:4996)
using namespace std;

typedef struct node
{
	double dis, price;
}station;

bool cmp(station a, station b)
{
	return a.dis < b.dis;
}


int main(void) {
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif
	
	double c, end_dis, average;
	int n;
	//cin >> c >> end_dis >> average >> n;
	scanf("%lf%lf%lf%d", &c, &end_dis, &average, &n);

	vector<station> v(n + 1);	

	for (int i = 0; i < n; i++)
	{
		//cin >> v[i].price >> v[i].dis;
		scanf("%lf%lf", &v[i].price, &v[i].dis);
	}

    //就是认为最后一战为1300，作为遍历的结尾
	v[n].dis = end_dis;
	v[n].price = 0;
	//排序你懂的
	sort(v.begin(), v.end(), cmp);

	if (v[0].dis != 0) {                                    //特判起点没有加油站 
		printf("The maximum travel distance = 0.00");
		return 0;
	}

	//车子从起点开始判断，第一站价格7.1，找下还有没有便宜的
	double oil = c * average, tank=0.0, sum=0.0;
	int  now = 0;				//now表示起点，从0号加油站开始，next表示我们要去的下一个加油站
	while (1)
	{
        //这个地方next一定要放在循环里面，如果next没有每次更新初始化，那么会无法退出某些情况的循环
        int next = -1;
		double min = 0x3fffffff;

		//开始找有没有便宜又可以到达的加油站，再便宜你也得到得了才行
		for (int i = now + 1; i <= n && v[now].dis + oil >= v[i].dis; i++)
		{
			if (v[i].price < min)
			{
				min = v[i].price;
				next = i;
				if (min < v[now].price)
					break;
			}
		}

		if (next == -1)
		{
			printf("The maximum travel distance = %.2f", v[now].dis +oil);
			break;
		}
		//说明可以到达
		double need_oil = (v[next].dis - v[now].dis) / average;
		if (min < v[now].price)
		{
			if (tank < need_oil)
			{
				sum += (need_oil - tank)*v[now].price;
				tank = 0;
			}
			else
				tank -= need_oil;
		}
		else
		{
			sum += (c - tank)*v[now].price;
			tank = c - need_oil;
		}
		now = next;
		if (next == n)
		{
			printf("%.2lf", sum); break;
		}
	}

	return 0;
}
