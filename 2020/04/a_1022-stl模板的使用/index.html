思路和坑点

这题说实话不难，唯一的难点在于怎么存数据。什么意思呢？就是给了m个查找信息，你如何从图书中找出你要的，那这个问题就演变成如何存放数据，使得我们可以很好的找到相应的id。那柳神是用了map<string,set<int>>来存储，你要高兴，也可以用unordered_map存储都一样。

AC代码

厚着脸皮借鉴一下柳神的代码

#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<map>
#include<algorithm> 
#include<vector>
#include<set>
#include<string> 
#pragma warning(disable:4996)
using namespace std;

map<string, set<int>> title,author,key,year,publisher;

void _find(map<string, set<int>> &tmp, string s)
{
	if (tmp.find(s) != tmp.end())
	{
		for (auto it = tmp[s].begin(); it != tmp[s].end(); it++)
			printf("%07d\n", *it);
	}
	else
		cout << "Not Found" << endl;
}


int main() {
#ifdef ONLINE_JUDGE    
#else    
	freopen("1.txt", "r", stdin);
#endif

	int n, m, id,temp;
	string t_title, t_author, t_key, t_puli, t_year;

	cin >> n;

	for (int i = 0; i < n; i++)
	{
		scanf("%d\n", &id);     //这地方不能cin+geiline，有一个换行
		getline(cin, t_title);
		title[t_title].insert(id);
		getline(cin, t_author);
		author[t_author].insert(id);
		while (cin >> t_key)
		{
			key[t_key].insert(id);
			char c = getchar();
			if (c == '\n')	break;		//这个地方不能给"\n"
		}
		getline(cin, t_puli);
		publisher[t_puli].insert(id);
		getline(cin, t_year);
		year[t_year].insert(id);
	}
	
	cin >> m;
	string s;
	for (int i = 0; i < m; i++)
	{
		scanf("%d: ", &temp);
		getline(cin, s);

		cout << temp << ": " << s << endl;
		if (temp == 1)
			_find(title, s);
		else if (temp == 2)
			_find(author, s);
		else if (temp == 3)
			_find(key, s);
		else if (temp == 4)
			_find(publisher, s);
		else if (temp == 5)
			_find(year, s);
	}
	return 0;
}
