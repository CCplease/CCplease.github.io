题目

样例输入

Input

2 1
01 1 02

Output

0 1

AC

这题本身没啥限制，只是算每层的叶子节点有几个，所以dfs和bfs都可以用，可爱建议大家都要学会，加深这两种遍历的印象

bfs

bfs大家应该熟悉的，就是层序遍历么，用一个队列来遍历每个节点，然后再遍历中，也就是while循环里面完成题目的相关要求

#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<queue>
#include<sstream>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

constexpr auto MAXN = 100;

vector<int> f[MAXN];

//layer表示每个节点的层数，比如题目样例中，2号节点，那么layer[2]==1，
//num就是每层有的叶子节点数，maxlevel表示最大层数，你也可以把num设为vector
//这样直接用vector.size()也是可以的
int layer[MAXN], num[MAXN], maxlevel = -1;

void bfs()
{
	queue<int> q;
	q.push(1);
	layer[1] = 0;
	while (!q.empty())
	{
		int index = q.front();
		q.pop();

        //以上的操作就是bfs常规操作
        //这个maxlevel要好好理解一下，求层数，因为一般都是在结构体里面加一个layer变量
		maxlevel = max(layer[index], maxlevel);

        //如果每一层的某一个节点的size==0，说明该节点就是叶子节点
		if (f[index].size() == 0)
			num[layer[index]]++;
        else
		{
            //否则说明该节点有叶子节点，那就继续压入队列，然后该节点层数要更新，
            //因为我们的num数组下标是层数，所以节点层数要更熟
            for (int i = 0; i < f[index].size(); i++)
		    {
			    q.push(f[index][i]);
			    layer[f[index][i]] = layer[index] + 1;
		    }
        }   
	}
}

int main()
{
#ifdef ONLINE_JUDGE                
#else    
	freopen("1.txt", "r", stdin);
#endif 
	
	int n, m;
	cin >> n >> m;

    //就是用二维变长数组存放一下，虽然是树(第一时间想到一维数组)，但是这题特殊
    //用图可以很好的完成题目的要求
	for (int i = 0; i < m; i++)
	{
		int node, tmp, temp;
		cin >> node >> tmp;
		for (int j = 0; j < tmp; j++)
		{
			cin >> temp;
			f[node].push_back(temp);
		}
	}
	
	bfs();

    //这个地方，优先输出层数为0的叶子节点个数，为了for循环方便一点(就是末尾不能有空格)
	cout << num[0];
	for (int i = 1; i<=maxlevel; i++)
	{
		cout << " " << num[i];
	}

	system("pause");
	return 0;
}

dfs

既然是dfs那么首先要想到递归的两个要素，一个是递归式，一个是递归边界。

  • 递归边界就是当前节点没有孩子节点了，也就是说这个节点就是叶子节点

  • 递归式就看当前节点有孩子节点，然后层数向下+1就行，因为只有当前节点是父节点才会执行递归

    #include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<queue>
#include<sstream>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

constexpr auto MAXN = 100;

vector<int> f[MAXN];
int num[MAXN], maxlevel = -1;

void dfs(int index,int layer)
{
	if (f[index].size() == 0)
	{
		num[layer]++;
		maxlevel = max(maxlevel, layer);
	}

	for (size_t i = 0; i < f[index].size(); i++)
	{
		dfs(f[index][i], layer + 1);
	}
}

int main()
{
#ifdef ONLINE_JUDGE                
#else    
	freopen("1.txt", "r", stdin);
#endif 
	
	int n, m;
	cin >> n >> m;

	for (int i = 0; i < m; i++)
	{
		int node, tmp, temp;
		cin >> node >> tmp;
		for (int j = 0; j < tmp; j++)
		{
			cin >> temp;
			f[node].push_back(temp);
		}
	}
	
	dfs(1, 0);

	cout << num[0];
	for (int i = 1; i <= maxlevel; i++)
		cout << " " << num[i];
	system("pause");
	return 0;
}
