题目

样例输入

Input_1

3
Joe M Math990112 89
Mike M CS991301 100
Mary F EE990830 95

Ouput_1

Mary EE990830
Joe Math990112
6

Input_2

1
Jean M AA980920 60

Output_2

Absent
Jean AA980920
NA

思路与坑点

这题就是标准可爱式偷懒，能用成员函数解决的绝不自己手写(并不倡导，还是提倡手写，即便出错了自己也知道错哪里)，这题就是三种情况，唯一要注意的是题目的意思是哪一个性别没有就加absent，absent并不是一定在第一行，不要理解错了。还有就是gender要用char，用string要出错。

AC代码

#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
#include<map>
#include<unordered_map>
#include<queue>
#include<set>
#include<stack>
#pragma warning (disable:4996)
using namespace std;

typedef struct node
{
	string name, id;
}info;

int main(void) {
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif

	map<int, info> man;
	map<int, info> women;

	int n;
	cin >> n;

	string name, id;
	char gender;
	int score;
	for (int i = 0; i < n; i++)
	{
		cin >> name >> gender >> id >> score;
		if (gender=='F')
		{
			women[score].name = name;
			women[score].id = id;
		}
		else if(gender=='M')
		{
			man[score].name = name;
			man[score].id = id;
		}
	}

    //3种情况，因为题目说了可能只有男没有女，或者只有女没有男
	if (man.size() != 0 && women.size() != 0)
	{
		cout << women.rbegin()->second.name << " " << women.rbegin()->second.id << endl;
		cout << man.begin()->second.name << " " << man.begin()->second.id << endl;
		cout << abs(women.rbegin()->first - man.rbegin()->first);
	}
	else if (man.size() == 0)
	{
		cout << women.rbegin()->second.name << " " << women.rbegin()->second.id << endl;
        cout << "Absent" << endl;
		cout << "NA";
	}
	else if (women.size() == 0)
	{
		cout << "Absent" << endl;
		cout << man.begin()->second.name << " " << man.begin()->second.id << endl;
		cout << "NA";
	}

	return 0;
}
