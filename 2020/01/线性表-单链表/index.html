线性表

　　关于线性表的概念，我用mooc数据结构课程老师讲的来解释
　　那么线性表的存储结构可以分为顺序存储和链式存储，本章我们只讲链式存储，为什么不讲顺序存储呢，因为顺序 存储时间复杂度和空间复杂度有点高，并且顺序存储能够做的事可以被链式代替，所以链式有典型性哦~

线性表-单链表

　　单链表，由很多很多存储空间，是通过线连在一起的表。在编程中，单链表又是通过节点+指针，将这些存储空间连接在 一起，可以方便的进行操作。

初始化链表

　　首先，既然是链表，那么节点的结构体和链表的结构体是不可少的
　　然后我们要初始化链表，可能会有人有疑惑，啥是初始化呀，请看图 　　红色框框就是初始化，因为是动态空间，所以得在堆区先来一个最原始得链表，及只有一个头节点得链表

#include<iostream>
#include<cstdlib>
using namespace std;

typedef struct LNode
{
	int data;
	struct LNode *next;
}Node, *PNode;

typedef struct List
{
	Node head;
	unsigned int num;

}*PList;

//初始化链表
void InitList(PList *list)	//这个地方参数，难点
{
	*list = (PList)malloc(sizeof(PList));
	if (*list)
	{
		(*list)->head.next = NULL;
	}
}

//求链表长度
unsigned GetListLength(PList list)
{
	if (list)
		return list->num;
	return 0;
}

//创建添加节点,同时也是插入，我们这里就写在一起了
void InserNode(PList list, int a, unsigned pos)
{
	if (pos > GetListLength(list))
		pos = GetListLength(list);
	PNode pNew = (PNode)malloc(sizeof(PNode));
	//接下来给新开辟的节点赋值
	if (pNew)
	{
		(pNew)->data = a;
		(pNew)->next = NULL;
	}
	//还需要创建一个指针用来查找
	PNode pNode = &list->head;
	while (pos--)
	{
		pNode = pNode->next;
	}
	pNew->next = pNode->next;
	pNode->next = pNew;
	list->num++;
}

//遍历链表
void TravseList(PList list)
{
	PNode pNode = (list->head).next;
	while (pNode)
	{
	
		cout << pNode->data << "\t";
		pNode = pNode->next;
	}
}

//删除有两种，根据位置删除和根据元素值删除
int DeletListToValue(PList list, int value)
{
	//1.先找一个新的指针指向头节点，找到我们要删的元素的前一个元素
	PNode pNode = &list->head;
	
	//2.判断移动找，流程是先判断是不是，然后移动，while结束后表示找到，在删除
	while (pNode->next && (pNode->next)->data != value)
	{
		pNode = pNode->next;
	}
	//3、到这里表示找到了那个要删除的值，但是指针指的是删除的前一个节点的next，也就是要删除的姐弟啊你
	if (pNode->next)
	{
		//用一个新的指针指向前一个节点，为啥要新的指针。首先删除特殊，需要用删除节点的前一个节点的指针，指向删除节点的后一个指针，那么
		//当你这样做后，你会发现，删除的节点丢失了，没有地方保存他，就掉队了
		PNode ptmp = pNode->next;
		cout << ptmp->data << endl;
		//修改链接
		pNode->next = ptmp->next;

		free(ptmp);
		//最重要的，list的数量--
		list->num--;
		return 1;
	}
	return 0;

}

int DeletListToPos(PList list, unsigned pos)
{
	//1.先找一个新的指针指向头节点，找到我们要删的元素的前一个元素
	PNode pNode = &list->head;

	if (pos > GetListLength(list))
	{
		printf("删除有误！\n");
		//cout << "删除有误，不对哟!" << endl;
		return 0;
	}
	unsigned i;
	for (i = 0; i < pos; i++)
	{
		pNode = pNode->next;
	}
	PNode p = pNode->next;
	pNode->next = pNode->next->next;
	free(p);
	list->num--;
	return 0;
}

int FindNode(PList list, unsigned pos)
{
	if (pos > GetListLength(list))
	{
		cout << "你想查找的位置不对" << endl;
		return 0;
	}
	PNode pNode = list->head.next;
	unsigned i = 0;
	for (i = 0; i < pos; i++)
	{
		pNode = pNode->next;
	}
	//找到了
	cout << endl << pNode->data << endl;
	return 0;
}

//销毁节点
void DestoryList(PList *plist)
{
	while (*plist)
		DeletListToPos(*plist, 0);
	free(*plist);
	*plist = NULL;
}

int main()
{
	PList p = NULL;
	InitList(&p);

	InserNode(p, 2, 0);
	InserNode(p, 12, 0);
	InserNode(p, 22, 0);
	InserNode(p, 32, 0);
	InserNode(p, 42, 0);
	TravseList(p);
	cout << endl << "--------------------------" << endl;
	cout << "删除元素12" << endl;
	DeletListToValue(p, 12);
	TravseList(p);
	cout << endl << "--------------------------" << endl;
	cout << "删除第二个元素" << endl;
	DeletListToPos(p, 1);
	TravseList(p);

	cout << endl << "--------------------------" << endl;
	cout << "查找元素" << endl;
	TravseList(p);
	FindNode(p, 1);
	DestoryList(&p);
	TravseList(p);

	getchar();
	return 0;
}
