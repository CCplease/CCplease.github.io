二叉树的存储

说实话二叉树的存储有很多种，你可以用静态的(就是顺序存储，用数组)，也可以用动态的(就是链表),我这里用动态的，这玩意我们也叫二叉链表 

typedef struct Node
{
	int data;
	struct Node *lchild;
	struct Node *rchild;
}binode,*bitree;

二叉树的建立

//其实这个Create函数，完全可以和插入写在一起，但是写代码的原则就是“低内聚，高耦合”，所以我们这里分开
//最后展现全部代码的时候 我是写在一起，两种方式，加深你们的理解
bitree Create(vector<int> &a, int n)		
{
	binode *root = NULL;
	for (int i = 0; i < n; i++)
	{
		_insert(root, a[i]);
	}
	return root;
}

二叉树节点的建立

binode *newnode(int num)
{
	binode *node = new binode;

	node->data = num;

	node->lchild = node->rchild = NULL;

	return node;

}

二叉树的插入

这里其实不是单纯的二叉树插入代码，是二叉搜索树的插入代码，小伙伴们看出来了吗！

void _insert(bitree &root, int value)
{
	if (root == NULL)
	{
		root = newnode(value);
		return;
	}
	if (value < root->data)
		_insert(root->lchild, value);
	else
		_insert(root->rchild, value);
}

二叉树的修改+查找

void search(binode *root, int value, int newdata)
{
	if (root == NULL)
		return;

	if (root->data == value)
		root->data = newdata;
	
	search(root->lchild, value, newdata);
	search(root->rchild, value, newdata);
}

二叉树的遍历

先序遍历(前序遍历)

先序遍历其实跟DFS(深度优先搜索)的结果是一样的，你也可以结合DFS加深对树遍历的学习。

void Inorder(binode *root)
{
	if (!root)
		return;
    visit(root);
	Inorder(root->lchild);
	Inorder(root->rchild);
}

中序遍历

void Inorder(binode *root)
{
	if (!root)
		return;
	Inorder(root->lchild);
	visit(root);            //这个地方你可以直接cout<<root->data;
	Inorder(root->rchild);
}

后序遍历

void Inorder(binode *root)
{
	if (!root)
		return;
	Inorder(root->lchild);
	Inorder(root->rchild);
    visit(root);
}

typedef struct LinkNode
{
    binode *data;
    struct LinkNode *next;
}LinkNode;

typedef struct
{
    LinkNode *front,*rear;
}LinkQueue;

void Levelorder(binode *root)
{
    LinkQueue Q;
    InitQueue(Q);
    bitree p;
    EnQueue(Q,root);
    while(!Is_empty(Q))
    {
        Dequeue(Q,p);
        visit(p);
        if(p->lchild != NULL)
            EnQueue(Q,p->lchild);
        else(p->rchild != NULL)
            EnQueue(Q,p->rchild);
    }
}

二叉树的删除

binode *minnode(binode *root)
{
	if (!root)
		return NULL;
	if (root->lchild == NULL)
		return root;
	else
		return minnode(root->lchild);
}

bool deletenode(bitree &root, int value)
{
	if (!root)
		return false;
	if (root->data == value)
	{
		if (!(root->lchild) && (root->rchild))
		{
			root->data = minnode(root->rchild)->data;
			return deletenode(root->rchild, root->data);
		}
		else
		{
            //这个写法的牛逼之处就是少了else if，所有你要删除的节点最终都会递归到这里被删除
            //不管你是叶子节点，还是只有左子树or右子树，还是左右都有
			binode * pdel = root;
			root = root->lchild ? root->lchild : root->rchild;
			free(pdel);
			return true;
		}
	}
	else if (root->data > value)
	{
		return deletenode(root->lchild, value);
	}
	else
	{
		return deletenode(root->rchild, value);
	}
}

这个方法你们可能看不太懂，影响不大，下面我写了一个通俗易懂的

//寻找最小
binode *findmin(binode * root){
    while(root->rchild!=NULL)
    {  
        root=root->rchild;
    }
    return root;
}

void deleteNode(binode * &root, int x)
{
	if (root == NULL) return;                              
	if (root->data == x) {                                  
		if (root->rchild == NULL && root->rchild == NULL)     
			root = NULL;
		else if (root->lchild != NULL) {                    
			Node *pre = findmax(root->lchild);           
			root->data = pre->data;                       
			deleteNode(root - lchild, pre->data);        
		}
		else {
			Node *next = findmin(root->rchild);           
			root->data = next->data;                       
			deleteNode(root - rchild, next->data);          
		}
	}
	else if (x > root->data)
		deleteNode(root->rchild, x);                     
	else
		deleteNode(root->lchild, x);                    
}

二叉树搜索某一个节点是否存在(不太重要)

bool searchnode(binode *root, int value)
{
	if (!root)
		return false;
	if (root->data == value)
		return true;
	if (root->data > value)
	{
		return searchnode(root->rchild, value);
	}
	else
	{
		return searchnode(root->lchild, value);
	}
}
