#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<vector>
#include<queue>
#include<sstream>
#include<algorithm>
using namespace std;

constexpr auto MAXN = 1001;

typedef struct node
{
	int id;
	int layer;
}Node;

vector<Node> v[MAXN];

int BFS(int s, int L)
{
	bool inq[1001] = { false };
	int numperson = 0;
	Node next;
	next.id = s;
	next.layer = 0;

	queue<Node> q;
	q.push(next);
	inq[next.id] = true;
	while (!q.empty())
	{
		Node topnode = q.front();
		q.pop();
		int first_num = topnode.id;
		for (size_t i = 0; i < v[first_num].size(); i++)
		{
			Node temp = v[first_num][i];
			temp.layer = topnode.layer + 1;
			if (inq[temp.id] == false && temp.layer <= L)
			{
				q.push(temp);
				inq[temp.id] = true;
				numperson++;
			}
		}
	}
	return numperson;
}
