A_1003-Dijkstra

三月 29, 2020 发布在 PAT, PAT-A

目录

  • 样例输入
    • Input:
    • Output:

样例输入

Input:

5 6 0 2
1 2 1 5 3
0 1 1
0 2 2
0 3 1
1 2 1
2 4 1
3 4 1

Output:

2 4

#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<vector>
#include<queue>
#include<sstream>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

constexpr auto MAXN = 501;
constexpr auto INF = 1000000000;

int G[MAXN][MAXN], weight[MAXN],d[MAXN],w[MAXN],num[MAXN];
int n,line,origin,destin;
void Dijkstra(int origin);
bool vis[MAXN] = { false };

int main()
{
#ifdef ONLINE_JUDGE                
#else    
	freopen("1.txt", "r", stdin);
#endif 

	cin >> n >> line >> origin >> destin;

	for (int i = 0; i < n; i++)
	{
		cin >> weight[i];
	}

	fill(G[0], G[0] + MAXN * MAXN, INF);
	int a1, a2, a3;
	for (int i = 0; i < line; i++)
	{
		cin >> a1 >> a2 >> a3;
		G[a1][a2] = G[a2][a1] = a3;
	}

	//这样，一个无向图就好了，那为啥一定要无向图，不可以是有向图呢
	Dijkstra(origin);

	cout << num[destin] << " " << w[destin];

	system("pause");
	return 0;
}

void Dijkstra(int origin)
{
	//记录一下最短的距离，来个数组d
	fill(d, d + MAXN, INF);
	//然后我们再初始化一下最大点权之和，因为因为要求输出救援小组数目之和
	//再初始化最短路径的条数，同样是题目要求
	memset(w, 0, sizeof(w));
	memset(num, 0, sizeof(num));

	d[origin] = 0;
	w[origin] = weight[origin];
	//这个1我说下，因为不管怎么样，顶点0-顶点2，有且至少有一条路，所以不肯能是0条
	num[origin] = 1;

	for (int i = 0; i < n; i++)
	{
		int u = -1, MIN = INF;
		for (int j = 0; j < n; j++)
		{
			//这个时候需要一个vis，来判断节点有没有被访问过
			if (vis[j] == false && d[j] < MIN)
			{
				u = j;
				MIN = d[j];
			}
		}

        //这个不解释了吧，就是说没找到相应的u，什么情况会执行return呢，就是全都遍历完了，所有节点都遍历完了，算是退出条件
		if (u == -1)	return;

		//程序能到这里，说明找到了一个节点是我们没访问过的，并且路径(d数组)是最小的,(路径好好理解一下，是从起点到节点的最短路径,也就是边权相加最小值)
		//这个时候解释一下u这个变量的作用，u表示我们当前的中介点的下标，也表示当前我们正在处理这个节点
		//这个标记的深层含义要理解，不仅仅是标记访问过，这表示这个点我哦们处理完了，也就是d[u]更新过了，以后我们再去用这个点可以直接用
		vis[u] = true;

		//u找到了，那么开始开放从u出发的边
		for (int v = 0; v < n; v++)
		{
			//什么叫开放u出发的边呢，首先，之前访问过的，不会访问，因为已经处理过了(就是d数组更新过了），所以肯定之后的边
			//举个例子，当u=1，那么就有1->0 and 1->2这两条边，1->0肯定不处理，因为d[0]更新过了，所以要处理1->2，所以这个循环
			//只有当v=2的时候，才是会往下面走的，其余直接跳过
			if (vis[v] == false && G[u][v] != INF)
			{
				//程序到这，我们接着u=1来解释，到这里，除了数组d要更新，题目要求要求点权和路径个数(也就是数组 w 和 num)
				//程序能走到这里，说明1->2这条边存在，也就是说2这个节点，从起点开始走，起码有一条路是通往2的，所以num[2]=0变为num[2]=1；
				//所以不管是if还是else if都是要处理num的，得让节点2的路径条数更新，同理w也是
				if (d[u] + G[u][v] < d[v])
				{
					d[v] = d[u] + G[u][v];
					w[v] = w[u] + weight[v];
					num[v] = num[u];
				}
				else if (d[u] + G[u][v] == d[v])
				{
					if (w[u] + weight[v] > w[v])
						w[v] = w[u] + weight[v];
					num[v] += num[u];
				}
				//那这里num[v]=num[u] and num[v]+=num[u];为什么这两个不一样呢.首先你要知道什么情况会进到if,什么时候会进到else if
				//进到if里面的，肯定是num[v]=0的情况，也就是第一次找到可以从起点到v节点的路径，那么如果到else if里面说明num[v]>=1
				//也就是说，之前已经找了通路，现在又有一条通路,所以直接+就行。

                //思考一下，为啥没有d[u]+G[u][v]>d[v]的情况，如果这个你懂了，那你就真的懂了，略略略~
			}
		}
	}
}
