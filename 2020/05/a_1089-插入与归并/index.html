A_1089-插入与归并

五月 15, 2020 发布在 PAT, PAT-A

目录

#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<cmath>
#include<algorithm>
#include<vector>
#include<string>
#include<map>
#include<unordered_map>
#include<queue>
#include<set>
#include<stack>
#pragma warning(disable:4996)
using namespace std;

/*这题，着实难，第一个难点，如何判断这个中间序列是插入排序还是归并排序的结果，我们采取判断他是否为插入，如果不是插入，那么
肯定是归并，题目没有说有其他情况，判断插入的方法就是，先用i指向中间序列中满足从左到右是从小到大顺序的最后一个下标，比如说例1，
i值得就是元素8那个位置，然后再将j指向从i+1开始，第一个不满足a[j] == b[j]的下标，如果j顺利到达了下标n，说明是插入排序，否则就是
归并。那么第二个难点是是如何输出下一轮排序的结果，依然是找规律，再下一次的序列是sort(a, a+i+2)，我也不知道为啥，别人是这样的。
那么归并的下一轮排序怎么办呢，归并就不用管中间序列了，把原序列按非递归开始排，排的时候要比较是否和中间序列相同，如果相同，那就再
排一次就退出，这个具体操作，代码会有详细注解*/

int main() {
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif
	
	ios::sync_with_stdio(false);

	int n, i, j;
	cin >> n;
	int a[100], b[100];
	for (i = 0; i < n; i++)
		cin >> a[i];
	for (i = 0; i < n; i++)
		cin >> b[i];

	//这里，i<n-1是因为，我们用当前的值和后面一个比，如果i是最后一个，那就超出界限了
	//这个外循环的意义就是为了 用i指向中间序列中满足从左到右是从小到大顺序的最后一个下标
	for (i = 0; i < n - 1 && b[i] <= b[i + 1]; i++);
	{
		//代码如果到这里，说明当前的i之前的值都是从小到大的，那么j=i+1，开始做做进一步的判断
		//如果，j=n，也就是后面的中序跟原序列一样，就是插入，否则就是归并
		for (j = i + 1; a[j] == b[j] && j < n; j++);
		{
			//代码到这里，标明是插入，然后用sort
			if (j == n)
			{
				cout << "Insertion Sort" << endl;
				sort(a, a + i + 2);
			}
			else
			{
				//到这里是归并
				cout << "Merge Sort" << endl;
				int k = 1, flag = 1;

				//这个while很重要，为啥是flag，flag是为了判断中序是否跟原序列一样了，如果一样了，flag=0，那么
				//在执行一次排序，又回到while，然后退出了循环，因为不满足条件
				while (flag)
				{
					flag = 0;
					for (i = 0; i < n; i++)
					{
						if (a[i] != b[i])
							flag = 1;
					}
					k = k * 2;
					for (i = 0; i < n / k; i++)
						sort(a + i * k, a + (i + 1) * k);
					sort(a + n / k * k, a + n);
				}
			}

		}
	}
	for (j = 0; j < n; j++)
	{
		if (j != 0) printf(" ");
		printf("%d", a[j]);
	}
	system("pause");
	return 0;
}
