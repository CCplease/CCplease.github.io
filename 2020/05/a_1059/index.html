思考与坑点

这题的素数表我是借鉴柳神的

本题就是求解质因数的过程，不同的是，还带有指数。

AC代码

#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
#include<queue>
#include<unordered_map>
#pragma warning(disable:4996)
using namespace std;

vector<int> prime(100000, 1);

void getprime()
{
	for (int i = 2; i * i < 10000; i++)
		for (int j = 2; j * i < 10000; j++)
			prime[j * i] = 0;
}

int main(void) 
{
#ifdef ONLINE_JUDGE
#else
	freopen("1.txt", "r", stdin);
#endif
	
	getprime();

	long long int a;
	cin >> a;

	printf("%d=", a);
	if (a == 1)
		printf("1");


	int flag = 0;
	for (int i = 2; a >= 2; i++)
	{
		int cnt = 0;
		while (prime[i] == 1 && a % i == 0)
		{
			cnt++;
			a /= i;
		}
		if (cnt)
		{
			if (flag)
				printf("*");
			flag = 1;
			if(cnt==1)	
				printf("%d", i);
			else if (cnt >= 2)
				printf("%d^%d", i,cnt);
		}

		if (a == 1)
			break;
	}

	return 0;
}
