思考与坑点

这题不要一上来直接结构体包含id、题目编号、分数，这样很明显不对，首先结构体肯定需要，需要包含总分total，一个包含关于题目分数的数组，一个题目通过的个数，还需要一个flag来标记他是不是无效的人员For those who has never submitted any solution that can pass the compiler, or has never submitted any solution, they must NOT be shown on the ranklist那么这样一个结构体所需要的就基本上清楚了，接下来注意以下几个问题：

  • 关于排名，标准的pat排名，样例的12225，碰到这种情况两个处理方法：1、在输出的时候用一个if判断加一个变量来处理这个问题。2、直接在结构体里面加一个rank，然后单独用一个for循环求出有效id的排名
  • For those who have the same rank, users must be sorted in nonincreasing order according to the number of perfectly solved problems. And if there is still a tie, then they must be printed in increasing order of their id's，读到这句话就应该知道标准的sort+cmp思路
  • 最后就是编译失败和未提交的区分，这里我们用-1和-2来区分，-1表示未提交，-2表示编译未通过，这样，我们在输出的时候就可以判断哪个输出0，哪个输出’-‘，我们给成绩的初始化时-1，表示最开始所有人都没提交过

AC代码

#include<iostream>
#include<cctype>
#include<cstring>
#include<cstdio>
#include<vector>
#include<unordered_map>
#include<sstream>
#include<map>
#include<queue>
#include<algorithm>
#pragma warning(disable:4996)
using namespace std;

typedef struct node
{
	int rank,id, sum,total;
	vector<int> score;		//放的是每门课的最高分
	int passnum=0;
	bool flag=false;
}Node;

vector<Node> v;

bool cmp(Node a, Node b)
{
	if (a.total != b.total)
		return a.total > b.total;
	else if (a.passnum != b.passnum)
		return a.passnum > b.passnum;
	else
		return a.id < b.id;
}
int main()
{
#ifdef ONLINE_JUDGE                
#else    
	freopen("1.txt", "r", stdin);
#endif 

	int n, k, m;
	cin >> n >> k >> m;

	v.resize(n + 1);
	for (int i = 1; i <= n; i++)
		//全部初始化为-1，表示都是没有提交过的意思，
		v[i].score.resize(k + 1, -1);

	vector<int> full_mark(k + 1);
	for (int i = 1; i <= k; i++)
		cin >> full_mark[i];

	for (int i = 0; i < m; i++)
	{
		int a, b, c;
		cin >> a >> b >> c;
		v[a].id = a;
		v[a].score[b] = max(v[a].score[b], c);

		if (c != -1)
			v[a].flag = true;
		else if (v[a].score[b] == -1)
			//说明提交过了，但是编译起未通过，所以表示为-2
			v[a].score[b] = -2;
	}

	//接下来要计算每个同学通过的题目数，注意，分数等于满分才算是通过,顺带算一下总分
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= k; j++)
		{
			if (v[i].score[j] != -1 && v[i].score[j] != -2)
				v[i].total += v[i].score[j];
			if (v[i].score[j] == full_mark[j])
				v[i].passnum++;
		}
	}

	//这样数据就基本上处理完了，开始根据题目要求排序
	sort(v.begin() + 1, v.end(), cmp);

	for (int i = 1; i <= n; i++)
	{
		v[i].rank = i;
		if (i != 1 && v[i].total == v[i - 1].total)
			v[i].rank = v[i - 1].rank;
	}

	//准备输出
	for (int i = 1; i <= n; i++)
	{
		if (v[i].flag == true)
		{
			printf("%d %05d %d", v[i].rank, v[i].id, v[i].total);
			for (int j = 1; j <= k; j++)
			{
				//3中情况
				if (v[i].score[j] != -1 && v[i].score[j] != -2)
					printf(" %d", v[i].score[j]);
				else if (v[i].score[j] == -1)
					printf(" -");
				else
					printf(" 0");
			}
			printf("\n");
		}
	}
	system("pause");
	return 0;
}
